# X.509 证书

提供`X509`证书的生成、保存和转换功能。

##### 示例

```csharp
using SharpDevLib;
using System.Security.Cryptography;
using System.Security.Cryptography.X509Certificates;

//生成RSA密钥对
var rsa = RSA.Create(2048);
//这里只是方便演示，实际ca,server,client都应该自己的密钥，csr也不一样
var privateKey = rsa.ExportPem(PemType.Pkcs1PrivateKey);

//生成证书序列号
var serialNumber = X509Helper.GenerateSerialNumber();
Console.WriteLine(serialNumber.HexStringEncode());
//5ce8f5164dbbc59f (示例)

//创建主体信息
var subject = new X509Subject("example.com")
{
    Country = "CN",
    Province = "Beijing",
    City = "Beijing",
    Organization = "Example Inc.",
    OrganizationalUnit = "IT Department"
};
Console.WriteLine(subject.Text());
//CN = example.com,C = CN,ST = Beijing,L = Beijing,O = Example Inc.,OU = IT Department

//创建证书签名请求
var csr = new X509CertificateSigningRequest(subject.Text(), privateKey);
var csrPem = csr.Export();
Console.WriteLine(csrPem);

//导入证书签名请求
var importedCsr = X509CertificateSigningRequest.Import(csrPem);
Console.WriteLine(importedCsr.Subject.Name);
//CN = example.com, C = CN, ST = Beijing, L = Beijing, O = Example Inc., OU = IT Department

//生成自签名CA证书
var caCert = X509Helper.GenerateSelfSignedCACert(privateKey, csr, serialNumber, 365, "My CA");
Console.WriteLine(caCert.Subject);
//CN = example.com, C = CN, ST = Beijing, L = Beijing, O = Example Inc., OU = IT Department
Console.WriteLine(caCert.Issuer);
//CN = example.com, C = CN, ST = Beijing, L = Beijing, O = Example Inc., OU = IT Department
Console.WriteLine(caCert.NotBefore);
//2026/3/8 8:00:00 (示例)
Console.WriteLine(caCert.NotAfter);
//2027/3/8 8:00:00 (示例)

//创建服务器证书的主体备用名称
var alternativeNames = new List<SubjectAlternativeName>
{
    new(SubjectAlternativeNameType.Dns, "example.com"),
    new(SubjectAlternativeNameType.Dns, "*.example.com"),
    new(SubjectAlternativeNameType.IP, "192.168.1.1"),
    new(SubjectAlternativeNameType.Uri, "https://example.com"),
    new(SubjectAlternativeNameType.Email, "admin@example.com")
};

//生成自签名服务器证书
var serverCert = X509Helper.GenerateSelfSignedServerCert(privateKey, csr, X509Helper.GenerateSerialNumber(), 365, alternativeNames, "My Server");
Console.WriteLine(serverCert.FriendlyName);
//My Server

//生成自签名客户端证书
var clientCert = X509Helper.GenerateSelfSignedClientCert(privateKey, csr, X509Helper.GenerateSerialNumber(), 365, "My Client");
Console.WriteLine(clientCert.FriendlyName);
//My Client

//生成自签名代码签名证书
var codeSigningCert = X509Helper.GenerateCodeSigningCert(privateKey, csr, X509Helper.GenerateSerialNumber(), 365, "My Code Signing");
Console.WriteLine(codeSigningCert.FriendlyName);
//My Code Signing

//使用CA证书颁发服务器证书
var issuerPrivateKey = privateKey;
var issuerCert = caCert;
var serverCert2 = X509Helper.GenerateServerCert(issuerPrivateKey, issuerCert, csr, X509Helper.GenerateSerialNumber(), 365, alternativeNames, "Server Cert");
Console.WriteLine(serverCert2.Issuer);
//CN = example.com, C = CN, ST = Beijing, L = Beijing, O = Example Inc., OU = IT Department

//使用CA证书颁发客户端证书
var clientCert2 = X509Helper.GenerateClientCert(issuerPrivateKey, issuerCert, csr, X509Helper.GenerateSerialNumber(), 365, "Client Cert");
Console.WriteLine(clientCert2.Issuer);
//CN = example.com, C = CN, ST = Beijing, L = Beijing, O = Example Inc., OU = IT Department

//使用CA证书颁发代码签名证书

Console.WriteLine(codeSigningCert2.Issuer);
//CN = example.com, C = CN, ST = Beijing, L = Beijing, O = Example Inc., OU = IT Department

//保存证书为DER格式
var derPath = "cert.der";
caCert.SaveDer(derPath);

//保存证书为PEM格式的CRT文件
var crtPath = "cert.crt";
caCert.SaveCrt(crtPath);

//保存证书为PFX格式
var pfxPath = "cert.pfx";
caCert.SavePfx(pfxPath, privateKey, "pfxPassword");

//将证书保存到流
caCert.SaveDer(File.Create("cert_from_stream.der"));

//将证书保存到流
caCert.SaveDer(File.Create("cert_from_stream.crt"));

Console.WriteLine(cert.GetRSAPrivateKey()?.ExportPem(PemType.Pkcs1PrivateKey));
```
